对象存储权限设置

# 样例

{
  "Id": "Policy1694501981111",
  "Statement": [
    {
      "Effect": "Allow",
      "Action": [
        "s3:GetObject",
        "s3:PutObject",
        "s3:ListBucket"
      ],
      "Principal": {
        "AWS": [
          "arn:aws:iam:::user/[用户名]"
        ]
      },
      "Resource": [
        "arn:aws:s3:::[桶名]",
        "arn:aws:s3:::[桶名]/[路径]/*"
      ]
    }
  ]
}
